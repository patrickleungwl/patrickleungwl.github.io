Using Mutexes

24 Dec 2015 on tech

Testing mutexes today. Mutexes can be used to ensure only one instance of an app runs at any one time.

 1using System;
 2using System.Threading;
 3
 4namespace TestMutex
 5{
 6    class MutexTester
 7    {
 8        private void LogMsg(string msg, params object[] args)
 9        {
10            Console.Out.Write("{0} ThreadId={1} ",
11                DateTime.Now.ToString("HH:MM:ss.fff"),
12                Thread.CurrentThread.ManagedThreadId);
13            Console.Out.WriteLine(msg, args);
14        }
15
16
17        // LongComputeTask
18        //
19        // Function run by background worker.
20
21        private void LongComputeTask()
22        {
23            LogMsg("LongComputeTask starting");
24
25            float percentDone = 0.0f;
26            for (int i = 0; i <10; i++)
27            {
28                Thread.Sleep(1000);
29                percentDone += 2.0f;
30                LogMsg("LongComputeTask {0:0.0f}% completed",
31                    percentDone);
32            }
33            LogMsg("LongComputeTask completed");
34
35        }
36
37
38        // RunTest
39        //
40        // Function run by main thread
41        // Main thread sets up the background thread
42        // and fires it off
43
44        public void RunTest()
45        {
46            // Acquire mutex here, ensure only 1 instance of this program
47            // run at any one time
48
49            var myMutex = new Mutex(false, "TestMutex");
50            myMutex.WaitOne();
51            LongComputeTask();
52            myMutex.ReleaseMutex();
53        }
54
55
56        static void Main(string[] args)
57        {
58            var tester = new MutexTester();
59            tester.RunTest();
60        }
61    }
62}
