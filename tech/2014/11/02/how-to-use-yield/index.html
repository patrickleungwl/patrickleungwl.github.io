How to use Yield

02 Nov 2014 on tech

Always wanted to be clear on how Yield works. This is a feature that traditional C or C++ do not have.

 1using System;
 2using System.Collections.Generic;
 3
 4
 5namespace TestYield
 6{
 7    class TestYield
 8    {
 9        // --------------------------------------------------
10        // AllOdds
11        //
12        // Yield provides an easy way to make an enumerator.
13        // Without Yield, you'd have to generate the entire list
14        // of items to iterate thru- and return that list to the
15        // user.
16        // --------------------------------------------------
17        
18        static public IEnumerable<int> AllOdds()
19        {
20            for (int i = 1; i < Int16.MaxValue; i++)
21            {
22                if (i % 2 == 1)
23                    yield return i;
24            }
25        }
26
27
28        // --------------------------------------------------
29        // Main
30        //
31        // Use the enumerator above- and can stop the enumerator
32        // from generating the next item by breaking at any time.
33        // --------------------------------------------------
34
35        static void Main(string[] args)
36        {
37            int count = 0;
38            foreach (int i in AllOdds())
39            {
40                Console.WriteLine("{0}", i);
41                if (++count > 10)
42                    break;
43            }
44        }
45    }
46}
