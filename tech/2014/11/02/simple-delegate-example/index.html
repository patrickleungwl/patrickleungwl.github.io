Simple Delegate Example

02 Nov 2014 on tech

Just playing with a simple delegate example. 

 1using System;
 2
 3namespace TestDelegate
 4{
 5    class TestDel
 6    {
 7        // Declare signature of operator methods
 8        delegate int OperatorSig(int a, int b);
 9
10        TestDel()
11        {
12        }
13
14
15        int Add(int a, int b)
16        {
17            return a + b;
18        }
19
20        int Subtract(int a, int b)
21        {
22            return a - b;
23        }
24 
25
26        static void Main(string[] args)
27        {
28            TestDel td = new TestDel();
29            OperatorSig myAdder = new OperatorSig(td.Add);
30            OperatorSig mySuber = new OperatorSig(td.Subtract);
31            Console.Out.WriteLine("Adder Result is {0}", myAdder(2, 3));
32            Console.Out.WriteLine("Suber Result is {0}", mySuber(2, 3));
33            Console.In.ReadLine();
34        }
35    }
36}
