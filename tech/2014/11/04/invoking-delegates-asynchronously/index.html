Invoking Delegates Asynchronously

04 Nov 2014 on tech

It’s a slow day at work. Let’s try an asynch delegate.

 1using System;
 2using System.Threading;
 3using System.Runtime.Remoting.Messaging;      // To access the AsyncResult type
 4
 5namespace TestAsyncDelegate
 6{
 7    class TestDel
 8    {
 9        delegate int OperatorSig(int a, int b);
10
11        TestDel()
12        {
13        }
14
15
16        // --------------------------------------------------------------
17        // Add
18        // 
19        // Random work method that takes an arbitrarily set 2.4 seconds
20        // to complete.
21        // --------------------------------------------------------------
22        
23        public int Add(int a, int b)
24        {
25            Console.Out.WriteLine("ThreadId={0}, Adding starts",
26                Thread.CurrentThread.ManagedThreadId );
27            Thread.Sleep(2400);
28            Console.Out.WriteLine("ThreadId={0}, Adding completed",
29                Thread.CurrentThread.ManagedThreadId );
30            return a + b;
31        }
32
33
34        // --------------------------------------------------------------
35        // ResultCallback
36        //
37        // This function gets called when the delegate method completes
38        // asynchronously in the background.
39        // We call EndInvoke to get the delegate results.
40        // --------------------------------------------------------------
41
42        public void ResultCallback(IAsyncResult iar)
43        {
44            AsyncResult ar = (AsyncResult)iar;                   // Get class object reference.
45            OperatorSig del = (OperatorSig)ar.AsyncDelegate;     // Get reference to delegate.
46            int result = del.EndInvoke(iar);
47            Console.Out.WriteLine("ThreadId={0}, Add result = {1}", 
48                Thread.CurrentThread.ManagedThreadId, result);
49        }
50
51
52        // --------------------------------------------------------------
53        // Main
54        // 
55        // --------------------------------------------------------------
56        
57        static void Main(string[] args)
58        {
59            TestDel td = new TestDel();
60
61            // create delegate and invoke to run asynchronously
62            OperatorSig myAdder = new OperatorSig(td.Add);
63            IAsyncResult res = myAdder.BeginInvoke(
64                                    10, 
65                                    5,
66                                    new AsyncCallback(td.ResultCallback), 
67                                    null);
68
69            // main thread keeps doing something else
70            while (!res.IsCompleted)
71            {
72                for (int i = 0; i < 10; i++)
73                {
74                    Console.Out.WriteLine("ThreadId={0}, Main thread working",
75                        Thread.CurrentThread.ManagedThreadId );
76                    Thread.Sleep(1000);
77                }
78            }
79        }
80    }
81}
