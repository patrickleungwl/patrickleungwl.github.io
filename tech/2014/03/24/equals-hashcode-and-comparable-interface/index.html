Equals Hashcode and Comparable Interface in Java

24 Mar 2014 on tech

When should a class overwrite equals() and hashCode()?

And when should a class implement the Comparable interface? A class should overwrite equals() when the class represents a value item- meaning ideas like dates, times, colors, or other unique representations of things. An object representing 2001-9-11 should equal another object also representing 2001-9-11, no? These two objects are equal despite being two different objects in memory. And each time you overwrite equals, you also have to overwrite hashCode()- because otherwise, each collection class that stores objects according to their hashCodes would not recognize them as being the same object.

Lastly, a class should implement the Comparable interface when you want to overwrite the default sorting. So if you want to sort the object backwards or add case insensitivity, you can do so in your own Comparable compareTo() routine.

  1package drills;
  2import java.util.ArrayList;
  3import java.util.Collections;
  4import java.util.HashSet;
  5import java.util.List;
  6import java.util.ListIterator;
  7import java.util.Set;
  8
  9
 10public final class MartianDateTime implements Comparable<MartianDateTime> {
 11
 12  // This test class demonstrates how to
 13  // 1. override equals()
 14  // 2. overrides hashCode()
 15  
 16  // Why override equals()?  When a class represents value, two instances of the same
 17  // class with the same value SHOULD be equal to each other. 
 18  // An example is a MartianDateTime class.  If two instances both represent 10AM, then
 19  // they should be considered equal.  (On the other hand, perhaps a static factory
 20  // constructor should ensure only one instance of the same datetime get created.)
 21  //
 22  // Why override hashCode()?  If you override equals() to mean two instances are 
 23  // equal, then they should also be considered equal when adding them to a 
 24  // Collection like hashSet or hashMap which uses an object's hashCode() to 
 25  // represent uniqueness.  
 26  
 27  private int martianHour;
 28  private int martianMinute;
 29  private int martianSecond;
 30  private int hashCode;
 31  
 32  // Make MartianDateTime an immutable class
 33  // seal it
 34  // make it constructed from a factory builder
 35  
 36  private MartianDateTime(int hour, int minute, int second) {
 37    martianHour = hour;
 38    martianMinute = minute;
 39    martianSecond = second;
 40    hashCode = 0;
 41  }
 42  
 43  // ---------------------------------------------------------------------
 44  // MartianDateTime
 45  // Static factory constructor 
 46  
 47  static public MartianDateTime BuildMartianDateTime(int hour, int minute, int second) {
 48    return new MartianDateTime(hour, minute, second);
 49  }
 50
 51  // ---------------------------------------------------------------------
 52  // toString()
 53  
 54  @Override
 55  public String toString() {
 56    return String.format("%d:%d:%d", martianHour, martianMinute, martianSecond);
 57  }
 58  
 59  // ---------------------------------------------------------------------
 60  // equals()
 61  //  return true if compare against self
 62  //  return false if compare against another type - use instanceof()
 63  //  cast to correct type
 64  //  return true if all significant fields compare equal
 65
 66  @Override
 67  public boolean equals(Object o) {
 68    if (this == o)
 69      return true;
 70    
 71    if (!(o instanceof MartianDateTime))
 72      return false;
 73    
 74    MartianDateTime other = (MartianDateTime)o;
 75    if (other.martianHour != this.martianHour || 
 76        other.martianMinute != this.martianMinute || 
 77        other.martianSecond != this.martianSecond ) 
 78      return false;
 79    
 80    return true;
 81  }
 82  
 83
 84  // ---------------------------------------------------------------------
 85  // hashCode
 86  // A good hashCode method should 
 87  // - unequal objects should return unequal hashcodes
 88  // - good hashing function should distribute objects across a range evenly
 89  //
 90  // simple recipe for a hashing function
 91  // 1. start with a prime number
 92  // 2. mix in contributions from each significant field
 93  
 94  @Override
 95  public int hashCode() {
 96    int tmp = hashCode;
 97    if (tmp==0) {
 98      tmp = 17;
 99      tmp = 31 * tmp + martianHour;
100      tmp = 31 * tmp + martianMinute;
101      tmp = 31 * tmp + martianSecond;
102      hashCode = tmp;
103    }
104    return hashCode;
105  }
106  
107
108  // ---------------------------------------------------------------------
109  // compareTo
110  // If this object is greater than comparedTo object, then return 1  
111  
112  @Override
113  public int compareTo(MartianDateTime o) {
114    
115    if (this.martianHour > o.martianHour)
116      return 1;
117    if ((this.martianHour == o.martianHour) &&
118      (this.martianMinute > o.martianMinute))
119      return 1;
120    if ((this.martianHour == o.martianHour) &&
121      (this.martianMinute == o.martianMinute) &&
122      (this.martianSecond > o.martianSecond))
123      return 1;
124    if ((this.martianHour == o.martianHour) &&
125      (this.martianMinute == o.martianMinute) &&
126      (this.martianSecond == o.martianSecond))
127      return 0;
128    
129    return -1;
130  }  
131
132  
133  // ---------------------------------------------------------------------
134  // test code
135  
136  public static void main(String[] args) {
137
138    // Construct 3 MartianDateTime instances
139    MartianDateTime dt1 = MartianDateTime.BuildMartianDateTime(10,8,8);
140    MartianDateTime dt2 = MartianDateTime.BuildMartianDateTime(10,8,9);
141    MartianDateTime dt3 = MartianDateTime.BuildMartianDateTime(10,8,8);
142    
143    // Instances with the same datetimes are equal
144    assert(!dt1.equals(dt2));
145    assert(dt1.equals(dt3));    
146  
147    // Should not be able to add two instances with the same "equal-ness"
148    // into the same hashset
149    Set<MartianDateTime> s = new HashSet<MartianDateTime>();
150    s.add(dt1);
151    s.add(dt2);
152    assert(s.contains(dt3));
153    
154    List<MartianDateTime> listMdt = new ArrayList<MartianDateTime>();
155    listMdt.add(dt1);
156    listMdt.add(dt2);
157    listMdt.add(dt3);
158    ListIterator<MartianDateTime> li = listMdt.listIterator();
159    System.out.println("Print list of MartianDateTimes");
160    while(li.hasNext()) {
161      MartianDateTime m=li.next();
162      if (m==null)
163        break;
164      System.out.println(m.toString());
165    }
166    
167    Collections.sort(listMdt);
168    ListIterator<MartianDateTime> sl = listMdt.listIterator();
169    System.out.println("Print list of sorted MartianDateTimes");
170    while(sl.hasNext()) {
171      MartianDateTime m=sl.next();
172      if (m==null)
173        break;
174      System.out.println(m.toString());
175    }
176    
177    System.out.println("Passed all tests");
178  }
179}
