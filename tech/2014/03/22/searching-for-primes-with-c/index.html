Searching for Primes with C

22 Mar 2014 on tech

As a software developer, I regularly do drills with different languages. This is rather analogous to how musicians play scales with their instructions. This past weekend, I brushed off some C cobwebs with a quick exercise to find prime numbers using the Sieve of Eratosthenes.

I remember from Jon Bentley’s “Programming Pearls” - back in the old days- when 64K RAM was considered a luxury, programmers came up with ingenius algorithms to conserve memory usage. I remember bit arrays were one fairly frugal data structures. I took it for a spin with the following prime search.

 10void show_bits(unsigned n)
 11{
 12    int i;
 13    for (i=(sizeof(int)*8)-1; i>0; i--)
 14    {
 15        (n&(1<<i)) ? putchar('1') : putchar('0');
 16    }
 17    printf( "\n" );
 18}

 25int is_bit_marked(unsigned *nums, unsigned u)
 26{
 27   unsigned bytenum = u / 32.0;
 28   unsigned offset = u % 32;
 29
 30   //printf( "is_bit_marked: %u, byte %u, bit %u\n", u, bytenum, offset );
 31   unsigned selected_byte = nums[bytenum];
 32   unsigned mask = 0x01;
 33   mask = mask << offset;   // rightshift offset times
 34
 35   //printf( "before mask: " );
 36   //show_bits(selected_byte);
 37
 38   //printf( "after mask:  " );
 39   selected_byte &= mask;   // and operation
 40   //show_bits(selected_byte);
 41   return ( selected_byte > 0 );
 42}

 45// given a number, find the byte and offset (the bit index) that
 46// represents the number.
 47//
 48void mark_bit(unsigned *nums, unsigned u)
 49{
 50   unsigned bytenum = u / 32.0;
 51   unsigned offset = u % 32;
 52
 53   //printf( "mark_bit: %u, byte %u, bit %u\n", u, bytenum, offset );
 54   unsigned mask = 0x01;
 55   mask = mask << offset;   // rightshift offset times
 56
 57   nums[bytenum] |= mask;
 58   //show_bits(selected_byte);
 59}

 62int main()
 63{
 64    unsigned x, y = 0;
 65    printf( "Length of unsigned %i\n\n", sizeof(unsigned));
 66
 67    // 32 bits per unsigned
 68    // to check for primes up to 1024*1024,
 69    // we'd need 1024*1024/32 unsigned variables
 70    // = 32,768 or 32k
 71    // 
 72    unsigned space_needed = 32768*16;
 73    unsigned nums[space_needed];
 74    memset(nums,0x00,space_needed);
 75
 76    // test our functions
 77    mark_bit(nums, 10);
 78    mark_bit(nums, 200);
 79    assert( is_bit_marked(nums, 2)==0 );
 80    assert( is_bit_marked(nums,24)==0 );
 81    assert( is_bit_marked(nums,300)==0 );
 82    assert( is_bit_marked(nums,10)==1 );
 83    assert( is_bit_marked(nums,200)==1 );
 84
 85    // continue on with main program
 86    memset(nums,0x00,space_needed);
 87    unsigned t=2;
 88    unsigned maxnum = 16*1024*1024;
 89
 90    for(t=2; t<maxnum; t++)
 91    {
 92        //printf("Checking %u\n", t);
 93        if (is_bit_marked(nums,t)==1)
 94            continue;
 95
 96        // mark every mutiple of t, but not t itself
 97        unsigned i=t+t;
 98        while(i<maxnum)
 99        {
100            mark_bit(nums, i);
101            i+=t;
102        }
103    }
104
105    int printed=0;
106    for(int t=2; t<maxnum; t++)
107    {
108        if (is_bit_marked(nums,t)==0)
109        {
110            printf( "%10u", t);
111            ++printed;
112        }
113
114        if (printed>4)
115        {
116            printed=0;
117            printf("\n");
118        }
119    }
120    printf("\n");
121}
