Using BackgroundWorker

20 Dec 2015 on tech

I’ve decided to take a deeper look at all available multi-threading options in .net 4.5. One of the simpler options is the BackgroundWorker. To investigate, I’ve setup a simple BackgroundWorker with progress and completion reporting.

  1using System;
  2using System.ComponentModel;
  3using System.Threading;
  4
  5namespace TestBackgroundWorker
  6{
  7    class TestBkGrdWorker
  8    {
  9        static void LogMsg(string msg, params object[] args)
 10        {
 11            Console.Out.Write("{0} ThreadId={1} ",
 12                DateTime.Now.ToString("HH:MM:ss.fff"),
 13                Thread.CurrentThread.ManagedThreadId);
 14            Console.Out.WriteLine(msg, args);
 15        }
 16
 17
 18        // RunTest
 19        //
 20        // Function run by main thread
 21        // Main thread sets up the background worker instance
 22        // and fires it off
 23
 24        public void RunTest()
 25        {
 26            BackgroundWorker bkWorker = new BackgroundWorker();
 27            bkWorker.WorkerReportsProgress = true;
 28
 29            // setup functions the background worker calls
 30            // to do work, report progress, and to report completion
 31            bkWorker.DoWork += bkWorker_DoWork;
 32            bkWorker.ProgressChanged += bkWorker_ProgressChanged;
 33            bkWorker.RunWorkerCompleted += bkWorker_RunWorkerCompleted;
 34
 35            AutoResetEvent myFlag = new AutoResetEvent(false);
 36
 37            // trigger background worker to run async
 38            bkWorker.RunWorkerAsync(myFlag);
 39
 40            LogMsg("Main thread working, BackgroundWorker busy={0}",
 41                bkWorker.IsBusy);
 42            myFlag.WaitOne();       // wait for someone to signal this flag
 43
 44            LogMsg("RunTest thread resumes control upon completion of background thread");
 45        }
 46
 47
 48        // bkWorker_ProgressChanged
 49        //
 50        // Report progress
 51
 52        void bkWorker_ProgressChanged(object sender, ProgressChangedEventArgs e)
 53        {
 54            LogMsg("Background worker update progress, {0:00.0}% done",
 55                e.ProgressPercentage);
 56        }
 57
 58        // bkWorker_RunWorkerCompleted
 59        //
 60        // Report work is completed
 61
 62        void bkWorker_RunWorkerCompleted(object sender, RunWorkerCompletedEventArgs e)
 63        {
 64            LogMsg("Background worker completed");
 65        }
 66
 67
 68        // bkWorker_DoWork
 69        //
 70        // Function run by background worker.
 71        // Random work method that takes an arbitrarily set 2 seconds
 72        // to complete, with a progress report every 200ms.
 73
 74        void bkWorker_DoWork(object sender, DoWorkEventArgs e)
 75        {
 76            LogMsg("Background worker starting");
 77            float percentDone = 0.0f;
 78            for (int i = 0; i < 10; i++)
 79            {
 80                Thread.Sleep(200);
 81                percentDone += 10.0f;
 82                BackgroundWorker bkwkr = sender as BackgroundWorker;
 83                if (bkwkr != null)
 84                {
 85                    bkwkr.ReportProgress((int)percentDone);
 86                }
 87            }
 88            LogMsg("Background worker completed");
 89
 90            AutoResetEvent myFlag = e.Argument as AutoResetEvent;
 91            if (myFlag != null)
 92            {
 93                myFlag.Set();
 94            }
 95        }
 96
 97
 98
 99        // Main
100        // Start up a background worker test instance
101
102        static void Main(string[] args)
103        {
104            TestBkGrdWorker testWkr = new TestBkGrdWorker();
105            testWkr.RunTest();
106        }
107    }
108}
109
