Using Tasks-2

27 Dec 2015 on tech

Testing with Tasks again. 

 1using System;
 2using System.Threading;
 3using System.Threading.Tasks;
 4
 5namespace TestTasks2
 6{
 7    class TasksTester
 8    {
 9        private void LogMsg(string msg, params object[] args)
10        {
11            Console.Out.Write("{0} ThreadId={1} ",
12                DateTime.Now.ToString("HH:MM:ss.fff"),
13                Thread.CurrentThread.ManagedThreadId);
14            Console.Out.WriteLine(msg, args);
15        }
16
17
18
19        private void DoWork()
20        {
21            LogMsg("Background worker starting");
22
23            float percentDone = 0.0f;
24            for (int i = 0; i < 10; i++)
25            {
26                Thread.Sleep(200);
27                percentDone += 10.0f;
28            }
29            LogMsg("Background worker completed");
30
31        }
32
33
34        // RunTest
35        //
36        // Function run by main thread
37        // Main thread sets up the background worker instance
38        // and fires it off
39
40        public void RunTest()
41        {
42            var myBkTask = new Task(DoWork);
43            var myBkTask2 = new Task(DoWork);
44            myBkTask.Start();
45            myBkTask2.Start();
46
47            LogMsg("Main thread worker waiting");
48            myBkTask.Wait();
49            myBkTask2.Wait();
50
51            LogMsg("Main thread worker completed");
52        }
53
54
55
56        // Main
57        // Start up a background worker test instance
58
59        static void Main(string[] args)
60        {
61            var testWkr = new TasksTester();
62            testWkr.RunTest();
63        }
64    }
65}
